35    private LevelState assertLevelState(String input, LevelState previous, Level expectedLevel, String expectedUntilContains) throws Exception {
36      LevelState back = LogProcessorUtils.levelStateOf(input, previous);
37      assertEquals(expectedLevel, back.level);
38      assertEquals(expectedUntilContains, back.untilContains);
39      return back;
40    }
