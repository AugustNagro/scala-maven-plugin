9       @Test
10      public void compare() throws Exception {
11          assertEquals(1, new VersionNumber("2.7.1").compareTo(new VersionNumber("1.0")));
12          assertEquals(1, new VersionNumber("2.7.1").compareTo(new VersionNumber("1.9")));
13          assertEquals(1, new VersionNumber("2.7.1").compareTo(new VersionNumber("2.0")));
14          assertEquals(1, new VersionNumber("2.7.1").compareTo(new VersionNumber("2.7")));
15          assertEquals(1, new VersionNumber("2.7.1").compareTo(new VersionNumber("2.7-rc")));
16          assertEquals(1, new VersionNumber("2.7.1").compareTo(new VersionNumber("2.7.0")));
17          assertEquals(0, new VersionNumber("2.7.1").compareTo(new VersionNumber("2.7.1")));
18          assertEquals(-1, new VersionNumber("2.7.1").compareTo(new VersionNumber("2.7.2-rc1")));
19          assertEquals(-1, new VersionNumber("2.7.1").compareTo(new VersionNumber("2.8")));
20          assertEquals(-1, new VersionNumber("2.7.1").compareTo(new VersionNumber("3.0")));
21      }

23      @Test
24      public void parse() throws Exception {
25          assertParseVN("2.7.1", 2, 7, 1, null);
26          assertParseVN("2.7", 2, 7, 0, null);
27          assertParseVN("2.7.1.RC", 2, 7, 1, ".RC");
28          assertParseVN("2.7.1.RC", 2, 7, 1, ".RC");
29          assertParseVN("2.7.RC", 2, 7, 0, ".RC");
30          assertParseVN("2.7-RC", 2, 7, 0, "-RC");
31          assertParseVN("2.7.1-SNAPSHOT", 2, 7, 1, "-SNAPSHOT");
32      }

34      @Test
35      public void compareMask() throws Exception {
36          assertEquals(1, new VersionNumberMask("2.7").compareTo(new VersionNumber("1.0")));
37          assertEquals(1, new VersionNumberMask("2.7").compareTo(new VersionNumber("1.9")));
38          assertEquals(1, new VersionNumberMask("2.7").compareTo(new VersionNumber("2.0")));
39          assertEquals(0, new VersionNumberMask("2.7").compareTo(new VersionNumber("2.7")));
40          assertEquals(0, new VersionNumberMask("2.7").compareTo(new VersionNumber("2.7-rc")));
41          assertEquals(0, new VersionNumberMask("2.7").compareTo(new VersionNumber("2.7.0")));
42          assertEquals(0, new VersionNumberMask("2.7").compareTo(new VersionNumber("2.7.1")));
43          assertEquals(0, new VersionNumberMask("2.7").compareTo(new VersionNumber("2.7.2-rc1")));
44          assertEquals(-1, new VersionNumberMask("2.7").compareTo(new VersionNumber("2.8")));

46          assertEquals(1, new VersionNumber("2.7-rc").compareTo(new VersionNumberMask("2.7")));
47          assertEquals(1, new VersionNumber("2.7.0").compareTo(new VersionNumberMask("2.7")));
48          assertEquals(1, new VersionNumber("2.7.1").compareTo(new VersionNumberMask("2.7")));
49          assertEquals(1, new VersionNumber("2.7.2-rc1").compareTo(new VersionNumberMask("2.7")));
50      }

52      private void assertParseVN(String str, int major, int minor, int bugfix, String modifier) {
53          VersionNumber v = new VersionNumber(str);
54          assertEquals("test major of " + str, major, v.major);
55          assertEquals("test minor of " + str, minor, v.minor);
56          assertEquals("test bugfix of " + str, bugfix, v.bugfix);
57          assertEquals("test modifier of " + str, modifier, v.modifier);
58      }
