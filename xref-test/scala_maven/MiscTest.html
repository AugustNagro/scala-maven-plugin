42      public void testClassworldSeftFirstStrategy() throws Exception {
43        ClassWorld w = new ClassWorld("zero", null);
44        ClassRealm rMojo = w.newRealm("mojo", getClass().getClassLoader());
45        Strategy s = new SelfFirstStrategy(w.newRealm("scalaScript", null));
46        ClassRealm rScript = s.getRealm();
47        rScript.setParentClassLoader(getClass().getClassLoader());
48        rScript.importFrom("mojo", MavenProject.class.getPackage().getName());
49        rScript.importFrom("mojo", MavenSession.class.getPackage().getName());
50        rScript.importFrom("mojo", Log.class.getPackage().getName());
51  
52  
53  
54        assertEquals(rScript, rScript.getStrategy().getRealm());
55        assertEquals(SelfFirstStrategy.class, rScript.getStrategy().getClass());
56  
57        File olderjar = new File(System.getProperty("user.home"), ".m2/repository/net/alchim31/maven/scala-maven-plugin/3.1.0/scala-maven-plugin-3.1.0.jar");
58        if (olderjar.exists()) {
59          System.out.println("found older jar");
60          rScript.addURL(olderjar.toURI().toURL());
61          String clname = "scala_maven.ScalaScriptMojo";
62          //assertNotSame(s.loadClass(clname), getClass().getClassLoader().loadClass(clname));
63          assertNotSame(rScript.loadClass(clname), getClass().getClassLoader().loadClass(clname));
64          assertSame(rMojo.loadClass(clname), getClass().getClassLoader().loadClass(clname));
65          assertSame(rScript.loadClass(MavenProject.class.getCanonicalName()), MavenProject.class);
66          assertSame(rScript.loadClass(MavenSession.class.getCanonicalName()), MavenSession.class);
67          assertSame(rScript.loadClass(Log.class.getCanonicalName()), Log.class);
68        }
