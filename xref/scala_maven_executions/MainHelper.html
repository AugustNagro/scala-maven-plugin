56      public static String toClasspathString(ClassLoader cl) throws Exception {
57          StringBuilder back = new StringBuilder();
58          List<String> cps = new LinkedList<String>();
59          appendUrltoClasspathCollection(cl, cps);
60          for(String cp : cps) {
61              if (back.length() != 0) {

64              back.append(cp);
65          }

69      public static void appendUrltoClasspathCollection(ClassLoader cl, Collection<String> classpath) throws Exception {
70          if (cl == null) {
71              cl = Thread.currentThread().getContextClassLoader();
72          }
73          while (cl != null) {
74              if (cl instanceof URLClassLoader) {
75                  URLClassLoader ucl = (URLClassLoader) cl;
76                  URL[] urls = ucl.getURLs();
77                  for (URL url : urls) {
78                      classpath.add(url.getFile());
79                  }
80              }
81              cl = cl.getParent();
82          }
83      }

90      private static String escapeArgumentForScalacArgumentFile(String arg) {
91          if(arg.matches(".*\\s.*")) {
92              return '"' + arg + '"';
93          }
94          return arg;

98       * UnEscapes arguments as necessary so the StringTokenizer for scala arguments pulls in filenames with spaces correctly.
99       * @param arg
100      * @return
101      */
102     private static String unescapeArgumentForScalacArgumentFile(String arg) {
103         if(arg.charAt(0) == '"' && arg.charAt(arg.length() -1) == '"') {
104             return arg.substring(1, arg.length() -1);
105         }
106         return arg;

117        //argFile.deleteOnExit();
118        final PrintWriter out = new PrintWriter(new BufferedWriter(new FileWriter(argFile)));
119        try {

125        }

149     /** Runs the main method of a java class */
150     public static void runMain(String mainClassName, List<String> args, ClassLoader cl) throws Exception {
151         if(cl == null) {
152             cl = Thread.currentThread().getContextClassLoader();
153         }
154         Class<?> mainClass = cl.loadClass(mainClassName);
155         Method mainMethod = mainClass.getMethod("main", String[].class);
156         int mods = mainMethod.getModifiers();
157         if(mainMethod.getReturnType() != void.class || !Modifier.isStatic(mods) || !Modifier.isPublic(mods)) {
158             throw new NoSuchMethodException("main");
159         }
160         String[] argArray = args.toArray(new String[args.size()]);
161 
162         //TODO - Redirect System.in System.err and System.out
163 
164         mainMethod.invoke(null, new Object[] {argArray});
165     }

167     public static String locateJar(Class<?> c) throws Exception {
168         final URL location;
169         final String classLocation = c.getName().replace('.', '/') + ".class";
170         final ClassLoader loader = c.getClassLoader();
171         if( loader == null ) {
172             location = ClassLoader.getSystemResource(classLocation);
173         } else {
174             location = loader.getResource(classLocation);
175         }
176         if( location != null ){
177             Pattern p = Pattern.compile( "^.*file:(.*)!.*$" ) ;
178             Matcher m = p.matcher( location.toString() ) ;
179             if( m.find() ) {
180                 return URLDecoder.decode(m.group( 1 ), "UTF-8") ;
181             }
182             throw new ClassNotFoundException( "Cannot parse location of '" + location + "'.  Probably not loaded from a Jar" ) ;
183         }
184         throw new ClassNotFoundException( "Cannot find class '" + c.getName() + " using the classloader" ) ;
185     }
