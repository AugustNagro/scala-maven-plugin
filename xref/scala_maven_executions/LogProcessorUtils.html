3   public class LogProcessorUtils {
4   
5       public enum Level {
6           ERROR, WARNING, INFO
7       }
8   
9       public static class LevelState {
10          public Level level = Level.INFO;
11          public String untilContains = null;
12      }
13  
14      public static LevelState levelStateOf(String line, LevelState previous) {
15          LevelState back = new LevelState();
16          String lineLowerCase = line.toLowerCase();
17          if (lineLowerCase.contains("error")) {
18              back.level = Level.ERROR;
19              if (lineLowerCase.contains(".scala")) {
20                  back.untilContains = "^";
21              }
22          } else if (lineLowerCase.contains("warn")) {
23              back.level = Level.WARNING;
24              if (lineLowerCase.contains(".scala")) {
25                  back.untilContains = "^";
26              }
27          } else if (previous.untilContains != null) {
28              if (!lineLowerCase.contains(previous.untilContains)) {
29                  back = previous;
30              } else {
31                  back.level = previous.level;
32                  back.untilContains = null;
33              }
34          }
35          return back;
36      }
