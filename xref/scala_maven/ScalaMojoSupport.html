538     @Override
539     public void execute() throws MojoExecutionException, MojoFailureException {
540         try {
541             String oldWay = System.getProperty("maven.scala.version");
542             if (oldWay != null) {
543                 getLog().warn("using 'maven.scala.version' is deprecated, use 'scala.version' instead");
544                 if (scalaVersion != null) {
545                     scalaVersion = oldWay;
546                 }
547             }
548 
549             oldWay = System.getProperty("maven.scala.displayCmd");
550             if (oldWay != null) {
551                 getLog().warn("using 'maven.scala.displayCmd' is deprecated, use 'displayCmd' instead");
552                 displayCmd = displayCmd || Boolean.parseBoolean(oldWay);
553             }
554             checkScalaVersion();
555             doExecute();
556         } catch (MojoExecutionException exc) {
557             throw exc;
558         } catch (MojoFailureException | RuntimeException exc) {
559             throw exc;
560         } catch (Exception exc) {
561             throw new MojoExecutionException("wrap: " + exc, exc);
562         }
563     }

569     VersionNumber findScalaVersion() throws Exception {
570         if (_scalaVersionN == null) {
571             String detectedScalaVersion = scalaVersion;
572             if (StringUtils.isEmpty(detectedScalaVersion)) {
573                 detectedScalaVersion = findScalaVersionFromDependencies();
574             }
575             if (StringUtils.isEmpty(detectedScalaVersion)) {
576                 if (!ScalaMojoSupport.POM.equals(project.getPackaging().toLowerCase())) {
577                     String error = getScalaOrganization() + ":" + SCALA_LIBRARY_ARTIFACTID
578                         + " is missing from project dependencies";
579                     getLog().error(error);
580                     throw new UnsupportedOperationException(error);
581                 }
582             } else {
583                 // grappy hack to retrieve the SNAPSHOT version without timestamp,...
584                 // because if version is -SNAPSHOT and artifact is deploy with uniqueValue then
585                 // the version
586                 // get from dependency is with the timestamp and a build number (the resolved
587                 // version)
588                 // but scala-compiler with the same version could have different resolved
589                 // version (timestamp,...)
590                 boolean isSnapshot = ArtifactUtils.isSnapshot(detectedScalaVersion);
591                 if (isSnapshot && !detectedScalaVersion.endsWith("-SNAPSHOT")) {
592                     detectedScalaVersion = detectedScalaVersion.substring(0,
593                         detectedScalaVersion.lastIndexOf('-', detectedScalaVersion.lastIndexOf('-') - 1)) + "-SNAPSHOT";
594                 }
595             }
596             if (StringUtils.isEmpty(detectedScalaVersion)) {
597                 throw new MojoFailureException("no scalaVersion detected or set");
598             }
599             if (StringUtils.isNotEmpty(scalaVersion)) {
600                 if (!scalaVersion.equals(detectedScalaVersion)) {
601                     getLog().warn(
602                         "scala library version define in dependencies doesn't match the scalaVersion of the plugin");
603                 }
604                 // getLog().info("suggestion: remove the scalaVersion from pom.xml");
605                 // //scalaVersion could be define in a parent pom where lib is not required
606             }
607             _scalaVersionN = new VersionNumber(detectedScalaVersion);
608         }
609         return _scalaVersionN;
610     }

616     // TODO refactor to do only one scan of dependencies to find version
617     private String findVersionFromDependencies(String groupId, String artifactId) {
618         String version = null;
619         for (Dependency dep : getDependencies()) {
620             if (groupId.equals(dep.getGroupId()) && artifactId.equals(dep.getArtifactId())) {
621                 version = dep.getVersion();
622             }
623         }
624         if (StringUtils.isEmpty(version)) {
625             List<Dependency> deps = new ArrayList<>();
626             deps.addAll(project.getModel().getDependencies());
627             if (project.getModel().getDependencyManagement() != null) {
628                 deps.addAll(project.getModel().getDependencyManagement().getDependencies());
629             }
630             for (Dependency dep : deps) {
631                 if (groupId.equals(dep.getGroupId()) && artifactId.equals(dep.getArtifactId())) {
632                     version = dep.getVersion();
633                 }
634             }
635         }
636         return version;
637     }

639     void checkScalaVersion() throws Exception {
640         String sv = findScalaVersion().toString();
641         if (StringUtils.isNotEmpty(scalaHome)) {
642             getLog().warn(String.format(
643                 "local scala-library.jar and scala-compiler.jar from scalaHome(%s) used instead of scala %s", scalaHome,
644                 sv));
645         }
646         if (checkMultipleScalaVersions) {
647             checkCorrectVersionsOfScalaLibrary(sv);
648         }
649     }

652      * this method checks to see if there are multiple versions of the scala library
653      * 
654      * @throws Exception
655      */
656     private void checkCorrectVersionsOfScalaLibrary(String scalaDefVersion) throws Exception {
657         getLog().debug("Checking for multiple versions of scala");
658         // TODO - Make sure we handle bad artifacts....
659         // TODO: note that filter does not get applied due to MNG-3236
660         VersionNumbermber">VersionNumber sv = new VersionNumber(scalaDefVersion);
661         VersionNumber requiredScalaVersion = StringUtils.isNotEmpty(scalaCompatVersion)
662             ? new VersionNumberMask(scalaCompatVersion)
663             : sv;
664         if (requiredScalaVersion.compareTo(sv) != 0) {
665             String msg = String.format("Scala library detected %s doesn't match scala.compat.version : %s", sv,
666                 requiredScalaVersion);
667             if (failOnMultipleScalaVersions) {
668                 getLog().error(msg);
669                 throw new MojoFailureException(msg);
670             }
671             getLog().warn(msg);
672         }
673         ProjectBuildingRequest request = project.getProjectBuildingRequest();
674         request.setProject(project);
675         checkArtifactForScalaVersion(requiredScalaVersion, dependencyTreeBuilder.buildDependencyGraph(request, null));
676     }

686 
687         CollectingDependencyNodeVisitor collectingVisitor = new CollectingDependencyNodeVisitor();
688         DependencyNodeVisitor firstPassVisitor = new FilteringDependencyNodeVisitor(collectingVisitor,
689             createScalaDistroDependencyFilter());
690         rootNode.accept(firstPassVisitor);
691 
692         DependencyNodeFilter secondPassFilter = new AncestorOrSelfDependencyNodeFilter(collectingVisitor.getNodes());
693         DependencyNodeVisitor filteredVisitor = new FilteringDependencyNodeVisitor(visitor, secondPassFilter);
694 
695         rootNode.accept(filteredVisitor);
696 
697         if (visitor.isFailed()) {
698             visitor.logScalaDependents();
699             if (failOnMultipleScalaVersions) {
700                 getLog().error("Multiple versions of scala libraries detected!");
701                 throw new MojoFailureException("Multiple versions of scala libraries detected!");
702             }
703             getLog().warn("Multiple versions of scala libraries detected!");
704         }
705     }

722     /**
723      * Get a {@link JavaMainCaller} used invoke a Java process. Typically this will
724      * be one of the Scala utilities (Compiler, ScalaDoc, REPL, etc.).
725      * <p>
726      * This method does some setup on the {@link JavaMainCaller} which is not done
727      * by merely invoking {@code new} on one of the implementations. Specifically,
728      * it adds any Scala compiler plugin options, JVM options, and Scalac options
729      * defined on the plugin.
730      *
731      * @param forkOverride
732      *            override the setting for {@link #fork}. Currently this should only
733      *            be set if you are invoking the REPL.
734      * @param mainClass
735      *            the JVM main class to invoke.
736      *
737      * @return a {@link JavaMainCaller} to use to invoke the given command.
738      */
739     final JavaMainCaller getScalaCommand(final boolean forkOverride, final String mainClass) throws Exception {
740         JavaMainCaller cmd = getEmptyScalaCommand(mainClass, forkOverride);
741         cmd.addArgs(args);
742         if (StringUtils.isNotEmpty(addScalacArgs)) {
743             cmd.addArgs(StringUtils.split(addScalacArgs, "|"));
744         }
745         addCompilerPluginOptions(cmd);
746         cmd.addJvmArgs(jvmArgs);
747         return cmd;
748     }

763     /**
764      * Get a {@link JavaMainCaller} used invoke a Java process. Typically this will
765      * be one of the Scala utilities (Compiler, ScalaDoc, REPL, etc.).
766      *
767      * @param mainClass
768      *            the JVM main class to invoke.
769      * @param forkOverride
770      *            override the setting for {@link #fork}. Currently this should only
771      *            be set if you are invoking the REPL.
772      *
773      * @return a {@link JavaMainCaller} to use to invoke the given command.
774      */
775     private JavaMainCaller getEmptyScalaCommand(final String mainClass, final boolean forkOverride) throws Exception {
776 
777         // If we are deviating from the plugin settings, let the user know
778         // what's going on.
779         if (forkOverride != fork) {
780             super.getLog().info("Fork behavior overridden");
781             super.getLog().info(String.format("Fork for this execution is %s.", String.valueOf(forkOverride)));
782         }
783 
784         // TODO - Fork or not depending on configuration?
785         JavaMainCaller cmd;
786         String toolcp = getToolClasspath();
787         if (forkOverride) {
788             // HACK (better may need refactor)
789             boolean bootcp = true;
790             if (args != null) {
791                 for (String arg : args) {
792                     bootcp = bootcp && !"-nobootcp".equals(arg);
793                 }
794             }
795             String cp = bootcp ? "" : toolcp;
796             bootcp = bootcp && !(StringUtils.isNotEmpty(addScalacArgs) && addScalacArgs.contains("-nobootcp"));
797             // scalac with args in files
798             // * works only since 2.8.0
799             // * is buggy (don't manage space in path on windows)
800             getLog().debug("use java command with args in file forced : " + forceUseArgFile);
801             cmd = new JavaMainCallerByFork(this, mainClass, cp, null, null, forceUseArgFile,
802                 toolchainManager.getToolchainFromBuildContext("jdk", session));
803             if (bootcp) {
804                 cmd.addJvmArgs("-Xbootclasspath/a:" + toolcp);
805             }
806         } else {
807             cmd = new JavaMainCallerInProcess(this, mainClass, toolcp, null, null);
808         }
809         return cmd;
810     }

895     protected List<File> getCompilerDependencies() throws Exception {
896         List<File> d = new ArrayList<>();
897         if (StringUtils.isEmpty(scalaHome)) {
898             for (Artifact artifact : getAllDependencies(getScalaOrganization(), SCALA_COMPILER_ARTIFACTID,
899                 findScalaVersion().toString())) {
900                 d.add(artifact.getFile());
901             }
902         } else {
903             for (File f : new File(scalaHome, "lib").listFiles()) {
904                 String name = f.getName();
905                 if (name.endsWith(".jar") && !name.contains("scala-library") && !name.contains("scala-compiler")) {
906                     d.add(f);
907                 }
908             }
909         }
910         return d;
911     }

920     private Set<Artifact> getAllDependencies(String groupId, String artifactId, String version) {
921         Set<Artifact> result = new HashSet<>();

961     /**
962      * Retrieves a list of paths to scala compiler plugins.
963      *
964      * @return The list of plugins
965      * @throws Exception
966      */
967     private Set<String> getCompilerPlugins() throws Exception {
968         Set<String> plugins = new HashSet<>();
969         if (compilerPlugins != null) {
970             Set<String> ignoreClasspath = new LinkedHashSet<>();
971             addCompilerToClasspath(ignoreClasspath);
972             addLibraryToClasspath(ignoreClasspath);
973             for (BasicArtifact artifact : compilerPlugins) {
974                 getLog().info("compiler plugin: " + artifact.toString());
975                 // TODO - Ensure proper scala version for plugins
976                 Set<String> pluginClassPath = new HashSet<>();
977                 // TODO - Pull in transitive dependencies.
978                 addToClasspath(artifact.groupId, artifact.artifactId, artifact.version, artifact.classifier,
979                     pluginClassPath, false);
980                 pluginClassPath.removeAll(ignoreClasspath);
981                 plugins.addAll(pluginClassPath);
982             }
983         }
984         return plugins;
985     }

987     private String findVersionFromPluginArtifacts(String groupId, String artifactId) {
988         for (Artifact art : pluginArtifacts) {
989             if (groupId.equals(art.getGroupId()) && artifactId.equals(art.getArtifactId())) {
990                 return art.getVersion();
991             }
992         }
993         throw new IllegalArgumentException(
994             "Could not locate artifact " + groupId + ":" + artifactId + " in plugin's dependencies");
995     }
