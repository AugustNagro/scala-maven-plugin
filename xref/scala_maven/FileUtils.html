7     /**
8      * @param canonical Should use CanonicalPath to normalize path (true => getCanonicalPath, false =&gt; getAbsolutePath)
9      * @see <a href="https://github.com/davidB/maven-scala-plugin/issues/50">#50</a>
10     */
11    public static String pathOf(File f, boolean canonical) throws Exception {

13    }

15    /**
