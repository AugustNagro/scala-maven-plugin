16   * Grossly hacky utility class which provides the fully unpacked OSGi classpath
17   * (ie. including nested .jars) when used in the context of Tycho and Maven 3.
18   * The use of reflection is required to avoid wiring in a dependency on either
19   * Tycho or Maven 3.
20   *
21   * @author miles.sabin
22   */
23  public class TychoUtilities {
24      private static final String TychoConstants_CTX_ECLIPSE_PLUGIN_CLASSPATH = "org.codehaus.tycho.TychoConstants/eclipsePluginClasspath";
25      private static final Method getContextValueMethod;
26      private static final Method getLocationsMethod;

28      static {
29          Method getContextValueMethod0 = null;
30          Method getLocationsMethod0 = null;
31          try {
32              final Class<?> mpClazz = MavenProject.class;
33              getContextValueMethod0 = AccessController.doPrivileged(new PrivilegedExceptionAction<Method>() {
34                  public Method run() throws Exception {
35                      Method m = mpClazz.getDeclaredMethod("getContextValue", String.class);
36                      m.setAccessible(true);
37                      return m;
38                  }
39              });

42              getLocationsMethod0 = AccessController.doPrivileged(new PrivilegedExceptionAction<Method>() {
43                  public Method run() throws Exception {

53          getLocationsMethod = getLocationsMethod0;

56      @SuppressWarnings("unchecked")
57      public static List<String> addOsgiClasspathElements(MavenProject project, List<String> defaultClasspathElements) {
58          if (getLocationsMethod == null) {
59              return defaultClasspathElements;
60          }
61  
62          List<Object> classpath = (List<Object>) getContextValue(project, TychoConstants_CTX_ECLIPSE_PLUGIN_CLASSPATH);
63          if (classpath == null || classpath.isEmpty())
64              return defaultClasspathElements;
65  
66          List<String> osgiClasspath = new ArrayList<String>();
67          for (Object classpathEntry : classpath) {
68              for (File file : getLocations(classpathEntry))
69                  osgiClasspath.add(file.getAbsolutePath());
70          }
71          osgiClasspath.addAll(defaultClasspathElements);
72          return osgiClasspath;
73      }
