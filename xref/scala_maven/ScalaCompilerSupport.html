90      @Override
91      protected void doExecute() throws Exception {
92          if (getLog().isDebugEnabled()) {
93              for(File directory : getSourceDirectories()) {
94                  getLog().debug(FileUtils.pathOf(directory, useCanonicalPath));
95              }
96          }
97          File outputDir = FileUtils.fileOf(getOutputDir(), useCanonicalPath);
98          File analysisCacheFile = FileUtils.fileOf(getAnalysisCacheFile(), useCanonicalPath);
99          int nbFiles = compile(getSourceDirectories(), outputDir, analysisCacheFile, getClasspathElements(), false);
100         switch (nbFiles) {
101             case -1:
102                 getLog().info("No sources to compile");
103                 break;
104             case 0:
105                 getLog().info("Nothing to compile - all classes are up to date");;
106                 break;
107             default:
108                 break;
109         }
110     }

112     protected int compile(List<File> sourceRootDirs, File outputDir, File analysisCacheFile, List<String> classpathElements, boolean compileInLoop) throws Exception, InterruptedException {
113         if (!compileInLoop && INCREMENTAL.equals(recompileMode)) {
114             // TODO - Do we really need this dupliated here?
115             if (!outputDir.exists()) {
116               outputDir.mkdirs();
117             }
118             // if not compileInLoop, invoke incrementalCompile immediately
119             return incrementalCompile(classpathElements, sourceRootDirs, outputDir, analysisCacheFile, compileInLoop);
120         }
121 
122         long t0 = System.currentTimeMillis();
123         LastCompilationInfo lastCompilationInfo = LastCompilationInfo.find(sourceRootDirs, outputDir);
124         if (_lastCompileAt < 0) {
125             _lastCompileAt = lastCompilationInfo.getLastSuccessfullTS();
126         }
127 
128         List<File> files = getFilesToCompile(sourceRootDirs, _lastCompileAt);
129 
130         if (files == null) {
131             return -1;
132         }
133 
134         if (files.size() < 1) {
135             return 0;
136         }
137         if (!outputDir.exists()) {
138             outputDir.mkdirs();
139         }
140         long t1 = System.currentTimeMillis();
141 
142         if (compileInLoop && INCREMENTAL.equals(recompileMode)) {
143             // if compileInLoop, do not invoke incrementalCompile when there's no change
144             int retCode = incrementalCompile(classpathElements, sourceRootDirs, outputDir, analysisCacheFile, compileInLoop);
145             _lastCompileAt = t1;
146             if (retCode == 1) {
147                 lastCompilationInfo.setLastSuccessfullTS(t1);
148             }
149             return retCode;
150         }
151 
152         getLog().info(String.format("Compiling %d source files to %s at %d", files.size(), outputDir.getAbsolutePath(), t1));
153         JavaMainCaller jcmd = getScalaCommand();
154         jcmd.redirectToLog();
155         if (!classpathElements.isEmpty()) jcmd.addArgs("-classpath", MainHelper.toMultiPath(classpathElements));
156         jcmd.addArgs("-d", outputDir.getAbsolutePath());
157         //jcmd.addArgs("-sourcepath", sourceDir.getAbsolutePath());
158         for (File f : files) {
159             jcmd.addArgs(f.getAbsolutePath());
160         }
161         if (jcmd.run(displayCmd, !compileInLoop)) {
162           lastCompilationInfo.setLastSuccessfullTS(t1);
163         }
164         else {
165             compileErrors = true;
166         }
167         getLog().info(String.format("prepare-compile in %d s", (t1 - t0) / 1000));
168         getLog().info(String.format("compile in %d s", (System.currentTimeMillis() - t1) / 1000));
169         _lastCompileAt = t1;
170         return files.size();
171     }

184     protected List<File> getFilesToCompile(List<File> sourceRootDirs, long lastSuccessfullCompileTime) throws Exception {
185         List<File> sourceFiles = findSourceWithFilters(sourceRootDirs);
186         if (sourceFiles.size() == 0) {
187             return null;
188         }
189 
190         // filter uptodate
191         // filter is not applied to .java, because scalac failed to used existing .class for unmodified .java
192         //   failed with "error while loading Xxx, class file '.../target/classes/.../Xxxx.class' is broken"
193         //   (restore how it work in 2.11 and failed in 2.12)
194         //TODO a better behavior : if there is at least one .scala to compile then add all .java, if there is at least one .java then add all .scala (because we don't manage class dependency)
195         List<File> files = new ArrayList<File>(sourceFiles.size());
196         if (_lastCompileAt > 0 || (!ALL.equals(recompileMode) && (lastSuccessfullCompileTime > 0))) {
197             ArrayList<File> modifiedScalaFiles = new ArrayList<File>(sourceFiles.size());
198             ArrayList<File> modifiedJavaFiles = new ArrayList<File>(sourceFiles.size());
199             ArrayList<File> allJavaFiles = new ArrayList<File>(sourceFiles.size());
200             for (File f : sourceFiles) {
201                 if (f.getName().endsWith(".java")) {
202                     allJavaFiles.add(f);
203                 }
204                 if (f.lastModified() >= lastSuccessfullCompileTime) {
205                     if (f.getName().endsWith(".java")) {
206                         modifiedJavaFiles.add(f);
207                     } else {
208                         modifiedScalaFiles.add(f);
209                     }
210                 }
211             }
212             if ((modifiedScalaFiles.size() != 0) || (modifiedJavaFiles.size() != 0)) {
213                 if ((modifiedScalaFiles.size() != 0) && MODIFIED_ONLY.equals(recompileMode)) {
214                     files.addAll(allJavaFiles);
215                     files.addAll(modifiedScalaFiles);
216                     notifyCompilation(files);
217                 } else {
218                     files.addAll(sourceFiles);
219                     notifyCompilation(sourceRootDirs);
220                 }
221             }
222         } else {
223             files.addAll(sourceFiles);
224             notifyCompilation(sourceRootDirs);
225         }
226         return files;
227     }

229     private void notifyCompilation(List<File> files) throws Exception {
230         if (notifyCompilation) {
231             for (File f : files) {
232                 getLog().info(String.format("%s:-1: info: compiling", FileUtils.pathOf(f, useCanonicalPath)));
233             }
234         }

237     private static class LastCompilationInfo {
238       static LastCompilationInfo find(List<File> sourceRootDirs, File outputDir) throws Exception {
239         StringBuilder hash = new StringBuilder();
240         for (File f : sourceRootDirs) {
241           hash.append(f.toString());
242         }
243         return new LastCompilationInfo(new File(outputDir.getAbsolutePath() + "." + hash.toString().hashCode() + ".timestamp"), outputDir);
244       }
245 
246       private final File _lastCompileAtFile;
247       private final File _outputDir;
248 
249       private LastCompilationInfo(File f, File outputDir) {
250         _lastCompileAtFile = f;
251         _outputDir = outputDir;
252       }
253 
254       long getLastSuccessfullTS() throws Exception {
255         long back =  -1;
256         if (_lastCompileAtFile.exists() && _outputDir.exists() && (_outputDir.list().length > 0)) {
257             back = _lastCompileAtFile.lastModified();
258         }
259         return back;
260       }
261 
262       void setLastSuccessfullTS(long v) throws Exception {
263         if (!_lastCompileAtFile.exists()) {
264             FileUtils.fileWrite(_lastCompileAtFile.getAbsolutePath(), ".");
265         }
266         _lastCompileAtFile.setLastModified(v);
267       }

274     @SuppressWarnings("unchecked")
275     protected int incrementalCompile(List<String> classpathElements, List<File> sourceRootDirs, File outputDir, File cacheFile, boolean compileInLoop) throws Exception, InterruptedException {
276         List<File> sources = findSourceWithFilters(sourceRootDirs);
277         if (sources.isEmpty()) {
278             return -1;
279         }
280 
281         if (incremental == null) {
282             File libraryJar = getLibraryJar();
283             File compilerJar = getCompilerJar();
284             List<File> extraJars = getCompilerDependencies();
285             extraJars.remove(libraryJar);
286             String sbtGroupId = SbtIncrementalCompiler.SBT_GROUP_ID;
287             String xsbtiArtifactId = SbtIncrementalCompiler.XSBTI_ARTIFACT_ID;
288             String compilerInterfaceArtifactId = SbtIncrementalCompiler.COMPILER_INTERFACE_ARTIFACT_ID;
289             String compilerInterfaceClassifier = SbtIncrementalCompiler.COMPILER_INTERFACE_CLASSIFIER;
290             String sbtVersion = findVersionFromPluginArtifacts(sbtGroupId, SbtIncrementalCompiler.COMPILER_INTEGRATION_ARTIFACT_ID);
291             File xsbtiJar = getPluginArtifactJar(sbtGroupId, xsbtiArtifactId, sbtVersion);
292             List<String> zincArgs = StringUtils.isEmpty(addZincArgs) ? new LinkedList<String>() : (List<String>) Arrays.asList(StringUtils.split(addZincArgs, "|"));
293             File interfaceSrcJar = getPluginArtifactJar(sbtGroupId, compilerInterfaceArtifactId, sbtVersion, compilerInterfaceClassifier);
294            	incremental = new SbtIncrementalCompiler(useZincServer, zincPort, libraryJar, compilerJar, extraJars, xsbtiJar, interfaceSrcJar, getLog(), zincArgs);
295         }
296 
297         classpathElements.remove(outputDir.getAbsolutePath());
298         List<String> scalacOptions = getScalaOptions();
299         List<String> javacOptions = getJavacOptions();
300         Map<File, File> cacheMap = getAnalysisCacheMap();
301 
302         try {
303             incremental.compile(project.getBasedir(), classpathElements, sources, outputDir, scalacOptions, javacOptions, cacheFile, cacheMap, compileOrder, toolchainManager.getToolchainFromBuildContext("jdk", session));
304         } catch (xsbti.CompileFailed e) {
305             if (compileInLoop) {
306                 compileErrors = true;
307             } else {
308                 throw e;
309             }
310         }
311 
312         return 1;
313     }

315     protected Map<File, File> getAnalysisCacheMap() {
316         HashMap<File, File> map = new HashMap<File, File>();
317         String scalaPluginKey = ((PluginDescriptor) getPluginContext().get("pluginDescriptor")).getPluginLookupKey();
318         for (MavenProject project1 : reactorProjects) {
319             Plugin plugin = project1.getPlugin(scalaPluginKey);
320             if (plugin != null) {
321                 Xpp3Dom configuration = (Xpp3Dom) plugin.getConfiguration();
322                 Xpp3Dom analysisCache = (configuration != null) ? configuration.getChild("analysisCacheFile") : null;
323                 File analysisCacheFile = (analysisCache != null) ? new File(analysisCache.getValue()) : defaultAnalysisCacheFile(project1);
324                 File classesDirectory = new File(project1.getBuild().getOutputDirectory());
325                 map.put(classesDirectory.getAbsoluteFile(), analysisCacheFile.getAbsoluteFile());
326                 Xpp3Dom testAnalysisCache = (configuration != null) ? configuration.getChild("testAnalysisCacheFile") : null;
327                 File testAnalysisCacheFile = (testAnalysisCache != null) ? new File(testAnalysisCache.getValue()) : defaultTestAnalysisCacheFile(project1);
328                 File testClassesDirectory = new File(project1.getBuild().getTestOutputDirectory());
329                 map.put(testClassesDirectory.getAbsoluteFile(), testAnalysisCacheFile.getAbsoluteFile());
330             }
331         }
332         return map;
333     }
