13  /**
14   * Run the Scala console with all the classes of the projects (dependencies and builded)
15   *
16   * @goal console
17   * @requiresDependencyResolution test
18   * @inheritByDefault false
19   * @requiresDirectInvocation true
20   * @executionStrategy once-per-session
21   */
22  public class ScalaConsoleMojo extends ScalaMojoSupport {
23  
24      /**
25       * The console to run.
26       *
27       * @parameter expression="${mainConsole}" default-value="scala.tools.nsc.MainGenericRunner"
28       * @required
29       */
30      protected String mainConsole;
31  
32      /**
33       * Add the test classpath (include classes from test directory), to the console's classpath ?
34       *
35       * @parameter expression="${maven.scala.console.useTestClasspath}" default-value="true"
36       * @required
37       */
38      protected boolean useTestClasspath;
39  
40      /**
41       * Add the runtime classpath, to the console's classpath ?
42       *
43       * @parameter expression="${maven.scala.console.useRuntimeClasspath}" default-value="true"
44       * @required
45       */
46      protected boolean useRuntimeClasspath;
47  
48      /**
49       * Path of the javaRebel jar. If this option is set then the console run
50       * with <a href="http://www.zeroturnaround.com/javarebel/">javarebel</a> enabled.
51       *
52       * @parameter expression="${javarebel.jar.path}"
53       */
54      protected File javaRebelPath;
55  
56      @Override
57      @SuppressWarnings("unchecked")
58      protected void doExecute() throws Exception {
59          //TODO - Many other paths uses the getScalaCommand()!!! We should try to use that as much as possibel to help maintainability.
60          VersionNumber scalaVersion = findScalaVersion();
61          String sv = scalaVersion.toString();
62          Set<String> classpath = new HashSet<String>();
63          addToClasspath("org.scala-lang", "scala-compiler", sv, classpath);
64          addToClasspath("org.scala-lang", "scala-library", sv, classpath);
65          if (new VersionNumber("2.9.0").compareTo(scalaVersion) <= 0) {
66            addToClasspath("org.scala-lang", "jline", sv, classpath);
67          } else {
68            addToClasspath("jline", "jline", "0.9.94", classpath);
69          }
70          classpath.addAll(project.getCompileClasspathElements());
71          if (useTestClasspath) {
72              classpath.addAll(project.getTestClasspathElements());
73          }
74          if (useRuntimeClasspath) {
75              classpath.addAll(project.getRuntimeClasspathElements());
76          }
77          String classpathStr = MainHelper.toMultiPath(classpath.toArray(new String[classpath.size()]));
78          JavaMainCaller jcmd = null;
79          List<String> list = new ArrayList<String>(args != null ? args.length + 3 : 3);
80          if(args != null) {
81              for(String arg : args) {
82                  list.add(arg);
83              }
84          }
85          list.add("-cp");
86          list.add(classpathStr);
87  
88          if(fork) {
89              getLog().warn("scala-maven-plugin cannot fork scala console!!  Running in process");
90          }
91  
92          jcmd = new JavaMainCallerInProcess(this, mainConsole, classpathStr, jvmArgs, list.toArray(new String[list.size()]));
93          //We need to make sure compiler plugins are sent into the interpreter as well!
94          addCompilerPluginOptions(jcmd);
95          if (javaRebelPath != null) {
96              if (!javaRebelPath.exists()) {
97                  getLog().warn("javaRevelPath '"+ javaRebelPath.getCanonicalPath()+"' not found");
98              } else {
99                  jcmd.addJvmArgs("-noverify", "-javaagent:" + javaRebelPath.getCanonicalPath());
100             }
101         }
102         jcmd.run(displayCmd);
103     }
