1   package scala_maven;
2   
3   import java.io.File;
4   import java.util.List;
5   
6   import org.apache.maven.model.Dependency;
7   import org.apache.maven.plugin.MojoExecutionException;
8   import org.apache.maven.plugin.MojoFailureException;
9   import org.apache.maven.plugins.annotations.LifecyclePhase;
10  import org.apache.maven.plugins.annotations.Mojo;
11  import org.apache.maven.plugins.annotations.Parameter;
12  import org.apache.maven.plugins.annotations.ResolutionScope;
13  
14  /**
15   * Compile Scala test source into test-classes. Corresponds roughly to
16   * testCompile
17   * in maven-compiler-plugin
18   *
19   */
20  @Mojo(name = "testCompile", defaultPhase = LifecyclePhase.TEST_COMPILE, requiresDependencyResolution = ResolutionScope.TEST, threadSafe = true)
21  public class ScalaTestCompileMojo extends ScalaCompilerSupport {
22  
23      /**
24       * Set this to 'true' to bypass unit tests entirely.
25       * Its use is NOT RECOMMENDED, but quite convenient on occasion.
26       *
27       */
28      @Parameter(property = "maven.test.skip")
29      protected boolean skip;
30  
31      /**
32       * The directory in which to place test compilation output
33       *
34       */
35      @Parameter(defaultValue="${project.build.testOutputDirectory}")
36      protected File testOutputDir;
37  
38      /**
39       * The directory in which to find test scala source code
40       *
41       */
42      @Parameter(defaultValue="${project.build.testSourceDirectory}/../scala")
43      protected File testSourceDir;
44  
45      @Override
46      public void execute() throws MojoExecutionException, MojoFailureException {
47          if (skip) {
48              return;
49          }
50          super.execute();
51      }
52  
53      /**
54       * Analysis cache file for incremental recompilation.
55       *
56       */
57      @Parameter(property="testAnalysisCacheFile", defaultValue="${project.build.directory}/analysis/test-compile")
58      protected File testAnalysisCacheFile;
59  
60      @Override
61      protected List<String> getClasspathElements() throws Exception {
62        List<String> back = project.getTestClasspathElements();
63        back.remove(project.getBuild().getTestOutputDirectory());
64        //back.add(getOutputDir().getAbsolutePath());
65        return back;
66      }
67  
68      @Override
69      protected List<Dependency> getDependencies() {
70          return project.getTestDependencies();
71      }
72  
73      @Override
74      protected File getOutputDir() throws Exception {
75          return testOutputDir.getAbsoluteFile();
76      }
77  
78      @Override
79      protected List<File> getSourceDirectories() throws Exception {
80          List<String> sources = project.getTestCompileSourceRoots();
81          String scalaSourceDir = testSourceDir.getAbsolutePath();
82          if(!sources.contains(scalaSourceDir)) {
83              sources.add(scalaSourceDir);
84          }
85          return normalize(sources);
86      }
87  
88      @Override
89      protected File getAnalysisCacheFile() throws Exception {
90          return testAnalysisCacheFile.getAbsoluteFile();
91      }
92  }
