31      public CheckScalaVersionVisitor(String projectVerison, Log log) {
32          this._version = new VersionNumber(projectVerison);
33          this._log = log;
34      }

36      public boolean isScalaDistroArtifact(Artifact artifact) {
37          return SCALA_DISTRO_GROUP.equalsIgnoreCase(artifact.getGroupId()) &&
38          SCALA_DISTRO_ARTIFACTS.contains(artifact.getArtifactId());
39      }
40      @Override
41      public boolean visit(DependencyNode node) {
42          //TODO - Do we care about provided scope?
43          Artifact artifact = node.getArtifact();
44          _log.debug("checking ["+artifact+"] for scala version");
45          //TODO - Handle version ranges???? does that make sense given scala's binary incompatability!
46          if(isScalaDistroArtifact(artifact) && artifact.getVersion() != null) {
47              VersionNumber originalVersion = new VersionNumber(artifact.getVersion());
48              if(originalVersion.compareTo(_version) != 0) {
49                  _failed = true;
50              }
51              //If this dependency is transitive, we want to track which artifact requires this...
52              if(node.getParent() != null) { //TODO - Go all the way up the parent chain till we hit the bottom....
53                  final Artifact parentArtifact = node.getParent().getArtifact();
54                  scalaDependentArtifactStrings.add(" " + StringUtil.makeArtifactNameString(parentArtifact) + " requires scala version: " + originalVersion);
55              }
56          } else {
57              //TODO - What now?
58          }
59          return !_failed;
60      }
61  
62      public boolean isFailed() {
63          return _failed;
64      }
65      public void logScalaDependents() {
66          _log.warn(" Expected all dependencies to require Scala version: " + _version);
