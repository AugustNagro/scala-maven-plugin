14  /**
15   * Ensures that all scala versions match the given version.
16   * @author JSuereth
17   *
18   */
19  public class CheckScalaVersionVisitor implements DependencyNodeVisitor {
20      private VersionNumber _version;
21      private boolean _failed = false;
22      private Log _log;
23      private String _scalaOrganization;
24  
25      private List<String> scalaDependentArtifactStrings = new ArrayList<String>();
26  
27      @Override
28      public boolean endVisit(DependencyNode node) {
29          return !_failed;
30      }
31  
32      public CheckScalaVersionVisitor(VersionNumber version, Log log,String scalaOrganization) {
33          this._version = version;
34          this._log = log;
35          this._scalaOrganization=scalaOrganization;
36      }
37  
38      public boolean isScalaDistroArtifact(Artifact artifact) {
39          return _scalaOrganization.equalsIgnoreCase(artifact.getGroupId()) &&
40          SCALA_DISTRO_ARTIFACTS.contains(artifact.getArtifactId());
41      }
42      @Override
43      public boolean visit(DependencyNode node) {
44          //TODO - Do we care about provided scope?
45          Artifact artifact = node.getArtifact();
46          _log.debug("checking ["+artifact+"] for scala version");
47          //TODO - Handle version ranges???? does that make sense given scala's binary incompatability!
48          if(isScalaDistroArtifact(artifact) && artifact.getVersion() != null) {
49              VersionNumber originalVersion = new VersionNumber(artifact.getVersion());
50              if(_version.compareTo(originalVersion) != 0) { //_version can be a VersionNumberMask
51                  _failed = true;
52              }
53              //If this dependency is transitive, we want to track which artifact requires this...
54              if(node.getParent() != null) { //TODO - Go all the way up the parent chain till we hit the bottom....
55                  final Artifact parentArtifact = node.getParent().getArtifact();
56                  scalaDependentArtifactStrings.add(" " + StringUtil.makeArtifactNameString(parentArtifact) + " requires scala version: " + originalVersion);
57              }
58          } else {
59              //TODO - What now?
60          }
61          return !_failed;
62      }
63  
64      public boolean isFailed() {
65          return _failed;
66      }
67      public void logScalaDependents() {
68          _log.warn(" Expected all dependencies to require Scala version: " + _version);
69          for(String dependString : scalaDependentArtifactStrings) {
70              _log.warn(dependString);
71          }
72      }
73  }
