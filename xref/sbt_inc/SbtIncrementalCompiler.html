34      public SbtIncrementalCompiler(File libraryJar, File reflectJar, File compilerJar, VersionNumber scalaVersion,
35          List<File> extraJars, File compilerBridgeJar, Log l, File cacheFile, CompileOrder compileOrder)
36          throws Exception {
37          this.compileOrder = compileOrder;
38          this.logger = new SbtLogger(l);
39          l.info("Using incremental compilation using " + compileOrder + " compile order");
40  
41          List<File> allJars = new ArrayList<>(extraJars);
42          allJars.add(libraryJar);
43          allJars.add(reflectJar);
44          allJars.add(compilerJar);
45  
46          ScalaInstance scalaInstance = new ScalaInstance( //
47              scalaVersion.toString(), // version
48              new URLClassLoader(
49                  new URL[] { libraryJar.toURI().toURL(), reflectJar.toURI().toURL(), compilerJar.toURI().toURL() }), // loader
50              ClasspathUtilities.rootLoader(), // loaderLibraryOnly
51              libraryJar, // libraryJar
52              compilerJar, // compilerJar
53              allJars.toArray(new File[] {}), // allJars
54              Option.apply(scalaVersion.toString()) // explicitActual
55          );
56  
57          ScalaCompiler scalaCompiler = new AnalyzingCompiler( //
58              scalaInstance, // scalaInstance
59              ZincCompilerUtil.constantBridgeProvider(scalaInstance, compilerBridgeJar), // provider
60              ClasspathOptionsUtil.auto(), // classpathOptions
61              new FromJavaConsumer<>(noop -> {
62              }), // onArgsHandler
63              Option.apply(null) // classLoaderCache
64          );
65  
66          compilers = ZincUtil.compilers( //
67              scalaInstance, //
68              ClasspathOptionsUtil.boot(), //
69              Option.apply(null), // javaHome
70              scalaCompiler);
71  
72          PerClasspathEntryLookup lookup = new PerClasspathEntryLookup() {
73              @Override
74              public Optional<CompileAnalysis> analysis(File classpathEntry) {
75                  return Optional.empty();
76              }
77  
78              @Override
79              public DefinesClass definesClass(File classpathEntry) {
80                  return Locate.definesClass(classpathEntry);
81              }
82          };
83  
84          analysisStore = AnalysisStore.getCachedStore(FileAnalysisStore.binary(cacheFile));
85  
86          setup = Setup.of( //
87              lookup, // lookup
88              false, // skip
89              cacheFile, // cacheFile
90              CompilerCache.fresh(), // cache
91              IncOptions.of(), // incOptions
92              new LoggedReporter(100, logger, pos -> pos), // reporter
93              Optional.empty(), // optionProgress
94              new T2[] {});
95      }
96  
97      private PreviousResult previousResult() {
98          Optional<AnalysisContents> analysisContents = analysisStore.get();
99          if (analysisContents.isPresent()) {
100             AnalysisContents analysisContents0 = analysisContents.get();
101             CompileAnalysis previousAnalysis = analysisContents0.getAnalysis();
102             MiniSetup previousSetup = analysisContents0.getMiniSetup();
103             return PreviousResult.of(Optional.of(previousAnalysis), Optional.of(previousSetup));
104         } else {
105             return PreviousResult.of(Optional.empty(), Optional.empty());
106         }
107     }
108 
109     public void compile(List<String> classpathElements, List<File> sources, File classesDirectory,
110         List<String> scalacOptions, List<String> javacOptions) {
111         List<File> fullClasspath = new ArrayList<>();
112         fullClasspath.add(classesDirectory);
113         for (String classpathElement : classpathElements) {
114             fullClasspath.add(new File(classpathElement));
115         }
116 
117         CompileOptions options = CompileOptions.of( //
118             fullClasspath.toArray(new File[] {}), // classpath
119             sources.toArray(new File[] {}), // sources
120             classesDirectory, scalacOptions.toArray(new String[] {}), // scalacOptions
121             javacOptions.toArray(new String[] {}), // javacOptions
122             100, // maxErrors
123             pos -> pos, // sourcePositionMappers
124             compileOrder, // order
125             Optional.empty() // temporaryClassesDirectory
126         );
127 
128         Inputs inputs = Inputs.of(compilers, options, setup, previousResult());
129 
130         CompileResult newResult = compiler.compile(inputs, logger);
131         analysisStore.set(AnalysisContents.create(newResult.analysis(), newResult.setup()));
132     }
