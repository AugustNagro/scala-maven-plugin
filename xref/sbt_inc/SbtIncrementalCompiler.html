43      public SbtIncrementalCompiler(boolean useZincServer, int zincPort, File libraryJar, File compilerJar, List<File> extraJars, File xsbtiJar, File interfaceJar, Log l, List<String> args) throws Exception {
44          this.log = l;
45          if (useZincServer) {
46              this.zinc = new ZincClient(zincPort);
47              if (zinc.serverAvailable()) {
48                  l.info("Using zinc server for incremental compilation");
49                  this.useServer = true;
50                  this.compilerJar = compilerJar;
51                  this.libraryJar = libraryJar;
52                  this.extraJars = extraJars;
53                  this.extraArgs = args;
54              } else {
55                  l.warn("Zinc server is not available at port " + zincPort + " - reverting to normal incremental compile");
56                  this.useServer = false;
57              }
58          }
59          if (!useServer) {
60              l.info("Using incremental compilation");
61              if (args.size() > 0) l.warn("extra args for zinc are ignored in non-server mode");
62              this.logger = new SbtLogger(l);
63              Setup setup = Setup.create(compilerJar, libraryJar, extraJars, xsbtiJar, interfaceJar, null, false);
64              if (l.isDebugEnabled()) Setup.debug(setup, logger);
65              this.compiler = Compiler.create(setup, logger);
66          }
67      }

96      private void zincCompile(File baseDir, List<String> classpathElements, List<File> sources, File classesDirectory, List<String> scalacOptions, List<String> javacOptions, File cacheFile, Map<File, File> cacheMap, String compileOrder) throws Exception {
97          List<String> arguments = new ArrayList<String>(extraArgs);
98          arguments.add("-log-level");
99          arguments.add(logLevelToString(log));
100         arguments.add("-scala-compiler");
101         arguments.add(compilerJar.getAbsolutePath());
102         arguments.add("-scala-library");
103         arguments.add(libraryJar.getAbsolutePath());
104         arguments.add("-scala-extra");
105         List<String> extraPaths = new ArrayList<String>();
106         for (File extraJar : extraJars) {
107             extraPaths.add(extraJar.getAbsolutePath());
108         }
109         arguments.add(MainHelper.toMultiPath(extraPaths));
110         if (!classpathElements.isEmpty()) {
111           arguments.add("-classpath");
112           arguments.add(MainHelper.toMultiPath(classpathElements));
113         }
114         arguments.add("-d");
115         arguments.add(classesDirectory.getAbsolutePath());
116         for (String scalacOption : scalacOptions) {
117             arguments.add("-S" + scalacOption);
118         }
119         for (String javacOption : javacOptions) {
120             arguments.add("-C" + javacOption);
121         }
122         arguments.add("-compile-order");
123         arguments.add(compileOrder);
124         arguments.add("-analysis-cache");
125         arguments.add(cacheFile.getAbsolutePath());
126         arguments.add("-analysis-map");
127         arguments.add(cacheMapToString(cacheMap));
128         for (File source : sources) {
129             arguments.add(source.getAbsolutePath());
130         }
131 
132         int exitCode = zinc.run(arguments, baseDir, System.out, System.err);
133 
134         if (exitCode != 0) {
135             xsbti.Problem[] problems = null;
136             throw new sbt.compiler.CompileFailed(arguments.toArray(new String[arguments.size()]), "Compile failed via zinc server", problems);
137         }
138     }

148     private String logLevelToString(Log l) {
149         if (l.isDebugEnabled()) return "debug";
150         else if (l.isInfoEnabled()) return "info";
151         else if (l.isWarnEnabled()) return "warn";
152         else if (l.isErrorEnabled()) return "error";
153         else return "info";

156     private String cacheMapToString(Map<File, File> cacheMap) throws Exception {
157         String analysisMap = "";
158         boolean addArgSeparator = false;
159         for (Map.Entry<File, File> entry : cacheMap.entrySet()) {
160             if (addArgSeparator) analysisMap += ANALYSIS_MAP_ARG_SEPARATOR;
161             analysisMap += entry.getKey().getAbsolutePath();
162             analysisMap += ANALYSIS_MAP_PAIR_SEPARATOR;
163             analysisMap += entry.getValue().getAbsolutePath();
164             addArgSeparator = true;
165         }
166         return analysisMap;
167     }
