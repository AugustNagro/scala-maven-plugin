91      @Override
92      protected CharSequence doRequest() throws Exception {
93          String yaml = toYaml(project).toString();
94          if (dumpYaml) {
95              new File(project.getBuild().getDirectory()).mkdirs();
96              FileUtils.fileWrite(project.getBuild().getDirectory() + "/project.yaml", "UTF-8", yaml);
97          }
98          StringBuilder back = new StringBuilder();
99          back.append(scs.sendRequestCreateOrUpdate(yaml));
100         if (compileAfterInit) {
101             back.append(scs.sendRequestCompile(project.getArtifactId()+"-"+project.getVersion(), true, true));
102         }
103         return back;
104     }
